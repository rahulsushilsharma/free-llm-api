You are given an integer N, print 2*W lines in the following manner- If N=4, then the pattern would be:
1 / 1

To achieve the specified pattern, you can use a loop to print 2*W lines, where W is the value of N. For each line, you can print a combination of numbers from 1 to N in ascending and descending order, separated by spaces. Here's a Python function to generate the pattern:

python
def print_pattern(N):
    W = N
    for i in range(1, N + 1):
        for j in range(1, W + 1):
            print(j, end=" ")
        for j in range(W, 0, -1):
            print(j, end=" ")
        print()
        W -= 1
        if W == 0:
            break

java
import java.util.*;

public class PatternPrint {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int N = scanner.nextInt();
        printPattern(N);
    }

    public static void printPattern(int N) {
        int num = 1;
        // Upper half of the pattern
        for (int i = 1; i <= N; i++) {
            for (int j = 1; j <= i; j++) {
                System.out.print(num++);
                if (j != i) {
                    System.out.print("*");
                }
            }
            System.out.println();
        }

        // Lower half of the pattern
        num = N * (N + 1) / 2 - N + 1;
        for (int i = N; i >= 1; i--) {
            for (int j = 1; j <= i; j++) {
                System.out.print(num++);
                if (j != i) {
                    System.out.print("*");
                }
            }
            System.out.println();
        }
    }
}
